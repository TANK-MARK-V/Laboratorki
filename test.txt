\section{СОЗДАНИЕ ФОРМУЛ}
\subsection{Включенная в текст формула}
\begin{center}
   \LARGE 
   Для чисел Фибоначи тождественно Кассини $F_{n+1}F_{n-1}-F_n^2=(-1)^2$ при $n>0$.
   
   Неравенства между средними: $\frac{n}{\frac{1}{a_1}+...\frac{+1}{a_n}} \le \sqrt[n]{a_1...a_n} \le \frac{a_1+...a_n}{n} \le \sqrt{\frac{a_n^2+...+a_n^2}{n}}$
   \end{center}
   \newpage
%---------------------------------------------------
% 2.2 ЗАДАНИЕ
%---------------------------------------------------
   \subsection{Выключная формула.}
   \begin{center}
   \LARGE
   Для чисел Фибоначи тождественно Кассини $$F_{n+1}F_{n-1}-F_n^2=(-1)^2$$ при $n>0$.
   
   Неравенства между средними: $$\frac{n}{\frac{1}{a_1}+...\frac{+1}{a_n}} \le \sqrt[n]{a_1...a_n} \le \frac{a_1+...a_n}{n} \le \sqrt{\left(\frac{a_n^2+...+a_n^2}{n}\right)}$$
\end{center}
